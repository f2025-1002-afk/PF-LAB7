
#include <iostream>
using namespace std;
 void printtable (int a,int b,int c){
    for(int i=a; i<=b;i++){
        cout << c << " x " << i << " = " << c * i << endl;
    }
}
int main() {
    int startnum ,endingnum,num;
      cout<<"enter starting number"<<endl;
      cin>>startnum;
      cout<<"enter ending number"<<endl;
      cin>>endingnum;
      cout<<"enter num"<<endl;
      cin>>num;
      
      printtable (startnum,endingnum,num);
      
    return 0;
}
